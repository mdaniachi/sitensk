function toggleTheme() {

  const html = document.documentElement;

  const currentTheme = html.getAttribute('data-theme');

  const newTheme = currentTheme === 'light' ? 'dark' : 'light';

  html.setAttribute('data-theme', newTheme);

}


function setLang(l) {

  document.documentElement.setAttribute('data-lang', l);

  document.documentElement.setAttribute('lang', l === 'pt' ? 'pt-BR' : 'en-US');

  document.querySelectorAll('.lb').forEach(b => b.classList.remove('on'));

  document.querySelector(`.lb[onclick="setLang('${l}')"]`).classList.add('on');

}

const aiWindow = document.getElementById('aiWindow');

const aiBody = document.getElementById('aiBody');

const aiInput = document.getElementById('aiInput');


function toggleChat() {

  aiWindow.classList.toggle('open');

  if(aiWindow.classList.contains('open')) {

    setTimeout(() => aiInput.focus(), 300);

  }

}


function handleAiChat(e) {

  e.preventDefault();

  const text = aiInput.value.trim();

  if(!text) return;

  

  const userMsg = document.createElement('div');

  userMsg.className = 'msg user';

  userMsg.textContent = text;

  aiBody.appendChild(userMsg);

  aiInput.value = '';

  scrollToBottom();

  

  const typingMsg = document.createElement('div');

  typingMsg.className = 'msg typing';

  typingMsg.textContent = 'Digitando...';

  aiBody.appendChild(typingMsg);

  scrollToBottom();

  

  setTimeout(() => {

    typingMsg.remove();

    const botMsg = document.createElement('div');

    botMsg.className = 'msg bot';

    botMsg.textContent = "Excelente! Entendi sua necessidade. Nossa equipe especialista em engenharia de produto pode marcar um papo rápido para mapear isso. Gostaria de receber um e-mail com horários?";

    aiBody.appendChild(botMsg);

    scrollToBottom();

  }, 1500);

}


function scrollToBottom() {

  aiBody.scrollTop = aiBody.scrollHeight;

}
